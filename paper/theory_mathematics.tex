\section{Hilbert Space Postulate}

Let

\[
\mathcal{H} = \bigotimes_{i=1}^{N} \mathcal{H}_i
\]

be a finite-dimensional Hilbert space representing $N$ elementary quantum degrees of freedom.

\textbf{Postulate.}

There exists a global pure quantum state

\[
\ket{\Psi} \in \mathcal{H}
\]

such that no background spacetime geometry is assumed.  
All geometric structures must emerge from correlations internal to $\ket{\Psi}$.

\section{Reduced Density Matrices}

For any subsystem $A \subset \{1,\dots,N\}$ we define

\[
\rho_A = \mathrm{Tr}_{\bar A}(\ket{\Psi}\bra{\Psi})
\]

The von Neumann entropy is

\[
S(A) = -\mathrm{Tr}(\rho_A \log \rho_A)
\]

\section{Mutual Information}

For two subsystems $i$ and $j$ we define

\[
I(i:j) = S(i) + S(j) - S(ij)
\]

Properties:

\begin{itemize}
\item $I(i:j) \ge 0$
\item symmetric
\item vanishes for uncorrelated systems
\end{itemize}

\section{Conditional Mutual Information}

For three subsystems

\[
\mathrm{CMI}(i:j|k)
\]

we define

\[
\mathrm{CMI}(i:j|k)
=
S(ik)+S(jk)-S(k)-S(ijk)
\]

CMI measures irreducible tripartite correlations.

\section{Entanglement Graph}

Define a weighted graph

\[
G=(V,E,w)
\]

where

\begin{itemize}
\item $V=\{1,\dots,N\}$
\item $w_{ij}=I(i:j)$
\end{itemize}

The adjacency matrix is

\[
W_{ij}=w_{ij}
\]

The degree matrix

\[
D_{ii}=\sum_j W_{ij}
\]

\section{Graph Laplacian}

The combinatorial Laplacian

\[
L=D-W
\]

is symmetric positive semi-definite.

The normalized Laplacian is

\[
L_{\text{norm}} = I - D^{-1/2}WD^{-1/2}
\]

\section{Heat Kernel}

Define

\[
K(t)=\mathrm{Tr}(e^{-tL})
\]

with eigenvalues $\lambda_k$:

\[
K(t)=\sum_k e^{-t\lambda_k}
\]

\section{Spectral Dimension}

The spectral dimension is

\[
d_s(t)=-2\frac{d\log K(t)}{d\log t}
\]

A plateau in $d_s(t)$ indicates emergent dimension.

\section{Information Distance}

Define

\[
d_{ij}=-\log\left(\frac{I(i:j)}{I_{\max}+\epsilon}\right)
\]

with small regularization $\epsilon$.

\section{Geometric Reconstruction}

Given the distance matrix $d_{ij}$, we reconstruct coordinates

\[
x_i \in \mathbb{R}^d
\]

via multidimensional scaling (MDS).

The embedding error is

\[
\epsilon_d =
\sqrt{
\frac{
\sum_{i<j}(\|x_i-x_j\|-d_{ij})^2
}{
\sum_{i<j}d_{ij}^2
}}
\]

The effective dimension is the minimal $d$ where $\epsilon_d$ stabilizes.

\section{Modular Hamiltonian}

For subsystem $A$:

\[
K_A = -\log \rho_A
\]

The modular flow is

\[
O(\tau) = e^{iK_A\tau} O e^{-iK_A\tau}
\]

\section{Spectral Statistics}

Let $\kappa_n$ be eigenvalues of $K_A$.

Define spacings

\[
\Delta_n = \kappa_{n+1}-\kappa_n
\]

Gap ratio

\[
r_n=
\frac{\min(\Delta_n,\Delta_{n+1})}
{\max(\Delta_n,\Delta_{n+1})}
\]

Mean gap ratio

\[
\langle r \rangle
\]

distinguishes Poisson vs chaotic spectra.

\section{Spectral Form Factor}

Define

\[
g_2(t)
=
\frac{1}{d_A^2}
\left|
\sum_{n=1}^{d_A}
e^{-it\tilde{\kappa}_n}
\right|^2
\]

The chaotic regime exhibits

\[
\text{dip} \rightarrow \text{ramp} \rightarrow \text{plateau}
\]

\section{Topological Modular Constant}

Define

\[
C = d_A g_2^{\mathrm{plateau}}
\]

The universal scaling

\[
g_2^{\mathrm{plateau}} \sim \frac{1}{d_A}
\]

is topology independent, but $C$ depends on interaction topology.

\section{Cut Entropy}

For region $A$ define

\[
S_{cut}(A)
=
\sum_{i\in A}\sum_{j\notin A}w_{ij}
\]

This measures information flow across the boundary.

\section{Ollivier–Ricci Curvature}

Define probability distribution

\[
\mu_i(j)=\frac{w_{ij}}{\sum_k w_{ik}}
\]

The curvature of edge $(i,j)$ is

\[
\kappa_{ij}
=
1
-
\frac{W_1(\mu_i,\mu_j)}{d(i,j)}
\]

where $W_1$ is the Wasserstein distance.

\section{Regional Curvature}

For region $A$

\[
\mathcal{R}(A)
=
\frac{1}{|E(A)|}
\sum_{(u,v)\in E(A)}
\kappa_{uv}
\]

\section{Horizon Score}

For region $A$

\[
Score_{BH}(A)
=
w_S z(S(A))
-
w_\phi z(\phi(A))
+
w_{int}z(internal(A))
\]

where

\[
\phi(A)
=
\frac{cut(A)}{\min(vol(A),vol(\bar A))}
\]

\section{Triangular Information Geometry}

Define

\[
T(i,j|k)
=
I(i:j)-I(i:k)-I(j:k)
\]

This score predicts

\[
\mathrm{CMI}(i:j|k)
\]

revealing triangular geometry in multipartite correlations.

\section{Limitations}

\begin{itemize}
\item finite $N$
\item absence of Lorentzian structure
\item no continuum limit proof
\end{itemize}

\section{Conclusion}

Bottom-Up quantum geometry reconstructs geometric observables directly from quantum information quantities without assuming spacetime structure.

Distance, dimension, curvature and horizon-like structures emerge from the entanglement graph of the quantum state.
